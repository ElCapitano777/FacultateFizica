double f(double x)
{
    return x*x-9;
}

{
    double a=0,b=10,c,delta=0.01,fa,fb;
        fa=f(a);
        fb=f(b);
    if (fa==0) 
    {
        cout<<a<<" este o solutie"<<endl;
    }
    if (fb==0) 
    {
        cout<<b<<" este o solutie"<<endl;
    }
    if ((fa*fb)<0) 
    while(true)
    {
        c=(a+b)/2;
        if (fabs(f(c))<delta) 
        {
            cout<<c<<" este solutie"<<endl;
            break;
        }
        if ((fa*f(c))<0) b=c;
        else a=c;
    }
    else cout<<"nu se poate reolva ecuatia";
