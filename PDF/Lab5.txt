#include <iostream>
#define n 5
using namespace std;

void print_vector(double x[])
{
    int i;
    for(i=0;i<n;i++)
        cout<<"a["<<i<<"]="<<x[i]<<endl;
}
void sortare_directa(double a[],double a_sortat[])
{
    int i,j=0,m;
    double maxim=-99999;
    for(i=0;i<n;i++)
        if(a[i]>maxim)maxim=a[i];
    while(j<n)
    {
        double minim=maxim+1;
        for(i=0;i<n;i++)
        if(a[i]<=minim)
            {
                minim=a[i];
                m=i;
            }
        a_sortat[j]=minim;
        a[m]=maxim+2;
        j++;
    }
}
int main()
{
    double a[n]={4.5,7.2,1,-3,-1.5}; double a_sortat[n];
    sortare_directa(a,a_sortat);
    print_vector(a_sortat);
}
