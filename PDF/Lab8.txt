#include <iostream>
#define n 3
using namespace std;

void afisare_matrice(double a[n+1][n+1])
{
    int i,j;
    for(i=0;i<n+1;i++)
    {
        for(j=0;j<n+1;j++)
        cout<<a[i][j]<<' ';
        cout<<endl;
    }
    cout<<endl;
}
void afisare_vector(double b[n+1])
{
    int i;
    for(i=0;i<n+1;i++)
        cout<<b[i]<<' ';
    cout<<endl<<endl;
}
void sist_triunghi(double a[n+1][n+1],double b[n+1],double x[n+1])
{
    int i,j;
    double suma;
    x[n+1]=b[n+1]/a[n+1][n+1];
    for(i=n+1;i>0;i--)
    {
        suma = 0;
        for(j=i+1;j<n+1;j++)
        suma=suma+a[i][j]*x[j];
    x[i]=(b[i]-suma)/a[i][i];
    }
}
void descompunere(double a[n+1][n+1],double b[n+1],double x[n+1])
{
    int i,j,k;
    double suma,l[n+1][n+1]={0},u[n+1][n+1]={0},y[n+1];
    for(i=1;i<=n+1;i++)
    l[i][i]=1;
        for(j=1;j<n+1;j++)
        {
            for(i=1;i<=j;i++)
    {
        suma=0;
        for(k=1;k<=j-1;k++)
            suma=suma+l[i][k]*u[k][j];
        u[i][j]=a[i][j]-suma;
    }
    for(i=j+1;i<n+1;i++)
     {
        suma=0;
        for(k=1;k<=j-1;k++)
            suma=suma+l[i][k]*u[k][j];
        l[i][j]=(a[i][j]-suma)/u[j][j];
    }
        }
    afisare_matrice(u);
    afisare_matrice(l);
    sist_triunghi(l,b,y);
    sist_triunghi(u,y,x);

}

int main()
{
    double a[n+1][n+1]={0,0,0,0,0,1,1,1,0,1,2,4,0,1,3,9};
    double b[n+1]={0,4,7,14},x[n+1]={0};
    descompunere(a,b,x);
    afisare_vector(x);
}
