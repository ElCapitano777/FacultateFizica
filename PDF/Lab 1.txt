{

//media a 10 nr aleatoare din interv [20;80]

#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main()
{
    int n[15],i;
    float s=0, medie;
    time_t t;
    
    srand((unsigned) time(&t));
    for(i=1;i<=10;i++)
    {
        n[i]=20+(rand()%61);
        s=s+n[i];
    }
    for(i=1;i<=10;i++)
    printf("%d  ", n[i]);
    printf("\n");
    medie=s/10;
    printf("media: %f",medie);
}

#include <stdlib.h>
#include <time.h>

int main()
{
    int i;

    srand((unsigned) time(&t));
    for(i=1;i<=10;i++)

        n=(200+(rand()%610));

        printf("%.1f  ", n);

//media a 10 nr aleatoare cu k zecimale din interv [20;80]

#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <math.h>

int main()
{
    int i, k, x;
    double n, s=0, medie;
    time_t t;
    printf("Cate zecimale?");
    scanf("%d", &k);
    srand((unsigned) time(&t));
    for(i=1;i<=10;i++)
    {
        x=61*pow(10,k);
        n=20*pow(10,k)+(rand()%x);
        n/=pow(10,k);
        printf("%f  ", n);
        s=s+n;
    }
    medie=s/10;
    printf("\n media: %f",medie);
}
